# 1. 安装依赖 & 构建 core

pnpm install
pnpm --filter @wemd/core build

# 2. 查看主题

WeMD>pnpm format:wechat -- --list-themes

跑渲染脚本
node ~/.cursor/skills/wemd-wechat-format/scripts/render-wechat.mjs \
 -i test.md -t modern-editorial -o test-wechat.html

# 3. 排版转换为HTML

WeMD>pnpm format:wechat -- -i .cursor/skills/wemd-wechat-format/fixtures/sample.md -t modern-editorial -o out.html

**粘贴: 浏览器打开 → 全选复制 → 公众号编辑器 Ctrl+V**
